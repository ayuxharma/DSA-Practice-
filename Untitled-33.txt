//insertion sort 

#include <iostream>
using namespace std ;

for (int i=1 ; i<n ; i++) {
    
     int temp = arr[i] ;
     int j = i - 1 ;

    for (int j= i-1 ; j>=0 ; j--) {
        
        if (arr[j] > temp) {
            //shift
            arr[j+1] = arr[j] ;
        }
        else { //ruk jao
            break ;
        }
        
    }
    arr[j+1] = temp ;
}
